    {

        File archive = createArchive( archiver, "zip" );

        org.apache.commons.compress.archivers.zip.ZipFile zf =
            new org.apache.commons.compress.archivers.zip.ZipFile( archive );

        int entryCount = 0;

        }

        assertEquals( 2, entryCount );

    }
