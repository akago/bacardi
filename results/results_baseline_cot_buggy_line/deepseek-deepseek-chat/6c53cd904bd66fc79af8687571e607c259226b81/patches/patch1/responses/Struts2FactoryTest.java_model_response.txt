            }
          },

    }
  }

  public void testStruts2Factory() {
    Struts2Factory s2Factory = new Struts2Factory();
    TestListener testListener =
        new TestListener(
            new AbstractModule() {

              @Provides
              @SuppressWarnings("unused")
              Date provideDate() {
                return TODAY;
              }
            });
