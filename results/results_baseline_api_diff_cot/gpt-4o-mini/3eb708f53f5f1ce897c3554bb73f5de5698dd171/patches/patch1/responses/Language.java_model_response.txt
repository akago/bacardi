```java
package com.google.cloud.translate;

import com.google.api.services.translate.v3.model.Language; // Updated import
import com.google.common.base.Function;
import com.google.common.base.MoreObjects;
import java.io.Serializable;
import java.util.Objects;

/**
 * Information about a language supported by Google Translation. Objects of this class contain the
 * language's code and the language name.
 *
 * @see <a href="https://cloud.google.com/translate/v2/discovering-supported-languages-with-rest">
 *     Discovering Supported Languages</a>
 * @see <a href="https://cloud.google.com/translate/docs/languages">Supported Languages</a>
 */
public class LanguageInfo implements Serializable { // Renamed class to avoid conflict with new API class

  private static final long serialVersionUID = 5205240279371907020L;
  static final Function<Language, LanguageInfo> FROM_PB_FUNCTION = // Updated to use new Language class
      new Function<Language, LanguageInfo>() {
        @Override
        public LanguageInfo apply(Language languagePb) {
          return LanguageInfo.fromPb(languagePb);
        }
      };

  private final String code;
  private final String name;

  private LanguageInfo(String code, String name) { // Updated constructor to match new class name
    this.code = code;
    this.name = name;
  }

  /** Returns the code of the language. */
  public String getCode() {
    return code;
  }

  /** Returns the name of the language. */
  public String getName() {
    return name;
  }

  @Override
  public String toString() {
    return MoreObjects.toStringHelper(this).add("code", code).add("name", name).toString();
  }

  @Override
  public final int hashCode() {
    return Objects.hash(code, name);
  }

  @Override
  public final boolean equals(Object obj) {
    if (obj == this) {
      return true;
    }
    if (obj == null || !obj.getClass().equals(LanguageInfo.class)) { // Updated to match new class name
      return false;
    }
    LanguageInfo other = (LanguageInfo) obj; // Updated to match new class name
    return Objects.equals(code, other.code) && Objects.equals(name, other.name);
  }

  static LanguageInfo fromPb(Language languagePb) { // Updated to use new Language class
    return new LanguageInfo(languagePb.getLanguageCode(), languagePb.getDisplayName()); // Updated method calls
  }
}
```
